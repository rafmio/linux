GDB

gcc -g input.c -o output
ключ -g - включить отладочную инфу в исполняемый код

(gdb) break main
(gdb) run 	# можно r
(gdb) next	# можно n
(gdb) print arr # в программе, написанной на Си можно вывести на печать текущее значение переменной


(gdb) x/4b arr  # выводит 4 байта после адреса, куда указывает переменная arr
(gdb) x/8b arr  # выводит 8 байт  после адреса, куда указывает переменная arr 

(gdb) set arr[1] = 'F' # изменить значение внyтри стека (заменить элемент массива arr[1] на 'F')

(gdb) set x/20x arr     # чтение 20 байт после адреса, на который указывает arr
			# учитывая, что изначально arr - был массив из 4х элементов, то у нас есть возможность
			# читать и изменять стек за пределами массива

(gdb) set *(0x7fffffffe144) = 0xff # изменить значение ячейки памяти 0x7fffffffe144 на 0xff
(gdb) set *(0x7fffffffe144 + 7) = 0xff # изменить значение ячейки памяти 0x7fffffffe144 + 7 на 0xff
